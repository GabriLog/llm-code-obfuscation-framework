    if (!Number.isInteger(n) || n < 0) throw new Error("Input must be a non-negative integer");
    else if (n === 0) return 1;
    else {
        let result = 1;
        for (let i = 1; i <= n; i++) {
            result *= i;
        }
        return result;
    }
