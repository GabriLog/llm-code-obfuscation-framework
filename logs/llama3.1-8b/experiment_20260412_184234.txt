  if (n % 2 === 0) return false;

  const sqrt = Math.sqrt(n);
  for (let i = 3; i <= sqrt; i += 2) {
    if (n % i === 0) return false;
  }
  return true;
}

function analyzeNumbers(nums) {
  let count = 0, sum = 0, max = -Infinity;

  nums.forEach((num) => {
    if (isPrime(num)) {
      count++;
      sum += num;
      max = Math.max(max, num);
    }
  });

  return { count, sum, average: sum / count || 0, max };
}

function main() {
  const limit = 100; 
  const nums = Array.from({ length: limit }, (_, i) => i + 1);

  console.log(analyzeNumbers(nums));
}
